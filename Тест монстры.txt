    <p id="result"></p>

<script>
const questions = [
    {
        question: "Зомби + дробовик",
        answers: [
            "2 в корпус",
            "2 в стане",
            "1 взрыв"
        ],
        correct: 0
    },
    {
        question: "Контролёр + пиротехника",
        answers: [
            "1 взрыв",
            "2 в корпус",
            "игнорирует"
        ],
        correct: 0
    },
    {
        question: "Бюрер + пиротехника",
        answers: [
            "игнорирует",
            "2 в корпус",
            "1 в корпус"
        ],
        correct: 0
    }
];

let current = 0;

function loadQuestion() {
    const q = questions[current];
    document.getElementById("question").innerText = q.question;
    const answersDiv = document.getElementById("answers");
    answersDiv.innerHTML = "";

    q.answers.forEach((answer, index) => {
        const btn = document.createElement("button");
        btn.innerText = answer;
        btn.onclick = () => checkAnswer(index, btn);
        answersDiv.appendChild(btn);
    });

    document.getElementById("result").innerText = "";
}

function checkAnswer(index, btn) {
    const q = questions[current];

    if (index === q.correct) {
        btn.classList.add("correct");
        document.getElementById("result").innerText = "✅ Верно!";
    } else {
        btn.classList.add("wrong");
        document.getElementById("result").innerText = "❌ Неверно!";
    }

    setTimeout(() => {
        current++;
        if (current < questions.length) {
            loadQuestion();
        } else {
            document.getElementById("question").innerText = "Тест завершён!";
            document.getElementById("answers").innerHTML = "";
        }
    }, 1000);
}

loadQuestion();
</script>
